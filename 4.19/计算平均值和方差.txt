#include <iostream>
#include <string>
#include <vector>

#include <iomanip>

using namespace std;

void twobitprint(double n)
{
    cout << fixed << setprecision(2) << n << ' ';
}



void var(int p[], int n, double ave)
{
	double s = 0.00;
	for (int i = 0; i < n; i++)
	{
		s += (p[i] - ave) * (p[i] - ave);
	}
	twobitprint(s/6);
}


int main() 
{
	int n; int sum = 0;
	double ave;
	int num[6];

	cin >> n;
	
	for(int j=0;j<n;j++)
	{
		for (int i = 0; i < 6; i++)
		{
			cin >> num[i];
			sum += num[i];			
		}
		double sumdouble = sum;
		ave = sumdouble / 6;
		twobitprint(ave);
		var(num, 6, ave);

		sum = 0; ave = 0; sumdouble = 0;
	}



	return 0;
}
